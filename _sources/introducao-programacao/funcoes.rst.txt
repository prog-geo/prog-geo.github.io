.. code-block:: python
    :linenos:
    :caption: Função para computar o fatorial de um número inteiro positivo.

    def Fatorial(num):
        produto = 1

        while(num > 0):
            produto = produto * num

            num = num - 1

        return produto


.. code-block:: python
    :linenos:
    :caption: Função para computar a distância euclidiana.

    import math

    def Distancia(x1, y1, x2, y2):
        dx = x2 - x1
        dy = y2 - y1

        d2 = dx**2 + dy**2

        d = math.sqrt(d2)

        return d
